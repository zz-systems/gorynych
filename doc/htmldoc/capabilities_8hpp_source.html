173  static constexpr raw_t fill_up_to(const capabilities capability)
174  {
175  auto value = to_underlying(capability);
176  uint64_t result = 0;
177 
178  for(size_t i = 0; i < ntz(value); i++)
179  {
180  result |= 1;
181  result <<= 1;
182  }
183 
184  return result;
