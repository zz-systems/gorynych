39  template<typename vreal_t>
40  std::enable_if_t<is_floating_point<vreal_t>::value, vreal_t> vsin(vreal_t val)
41  {
42  vreal_t q = (val * constants<vreal_t>::M_1_PI).floor();
43  zint iq = q;
44 
45  // when performance > precision
46  if(dispatcher::use_fast_float)
47  {
48  val = vfmadd(q, -constants<vreal_t>::M_PI, val);
49  }
50  else
51  {
52  val = vfmadd(q, -constants<vreal_t>::PI4_A * 4, val);
53  val = vfmadd(q, -constants<vreal_t>::PI4_B * 4, val);
54  val = vfmadd(q, -constants<vreal_t>::PI4_C * 4, val);
55  val = vfmadd(q, -constants<vreal_t>::PI4_D * 4, val);
56  }
57 
58  vreal_t s = val * val;
59 
60  val = (-val)
61  .when(iq & 1)
62  .otherwise(val);
63 
64  vreal_t u;
65 
66  // for single precision
67  if(is_float<vreal_t>::value)
68  {
69  u = 2.6083159809786593541503e-06f;
70  u = vfmsub(u, s, 0.0001981069071916863322258f);
71  u = vfmadd(u, s, 0.00833307858556509017944336f);
72  u = vfmsub(u, s, 0.166666597127914428710938f);
73  }
74  // for double precision
75  else if(is_double<vreal_t>::value)
76  {
77  u = - 7.97255955009037868891952e-18;
78  u = vfmadd(u, s, 2.81009972710863200091251e-15);
79  u = vfmsub(u, s, 7.64712219118158833288484e-13);
80  u = vfmadd(u, s, 1.60590430605664501629054e-10);
81  u = vfmsub(u, s, 2.50521083763502045810755e-08);
82  u = vfmadd(u, s, 2.75573192239198747630416e-06);
83  u = vfmsub(u, s, 0.000198412698412696162806809);
84  u = vfmadd(u, s, 0.00833333333333332974823815);
85  u = vfmsub(u, s, 0.166666666666666657414808);
86  }
87 
88 // u = s * u * val + val;
89  u = vfmadd(s, u * val, val);
90 
91  u = vreal_t::quiet_NaN()
92  .when(val.is_infinite())
93  .otherwise(u);
94 
95  return u;
96  }

247  template<typename vreal_t>
248  std::enable_if_t<is_floating_point<vreal_t>::value, vreal_t> vatan2(vreal_t y, vreal_t x)
249  {
250  auto q = vreal_t(2)
251  .when(x < 0)
252  .otherwise(0);
253 
254  x = x.abs();
255 
256  auto mask = y > x;
257  auto x0 = x, y0 = y;
258 
259  x = y0.when(mask).otherwise(x0);
260  y = (-x0).when(mask).otherwise(y0);
261  q += vreal_t(1.0).when(mask).otherwise(0);
262 
263  auto s = y / x;
264  auto t = s * s;
265 
266  vreal_t u;
267 
268  // for single precision
269  if(is_float<vreal_t>::value)
270  {
271  u = 0.00282363896258175373077393f;
272  u = vfmsub(u, t, 0.0159569028764963150024414f);
273  u = vfmadd(u, t, 0.0425049886107444763183594f);
274  u = vfmsub(u, t, 0.0748900920152664184570312f);

277  u = vfmadd(u, t, 0.199926957488059997558594f);
278  u = vfmsub(u, t, 0.333331018686294555664062f);
279  }

282  {
283  u = - 1.88796008463073496563746e-05;
284  u = vfmadd(u, t, 0.000209850076645816976906797);

289  u = vfmsub(u, t, 0.0254517624932312641616861);
290  u = vfmadd(u, t, 0.0337852580001353069993897);
291  u = vfmsub(u, t, 0.0407629191276836500001934);
292  u = vfmadd(u, t, 0.0466667150077840625632675);
293  u = vfmsub(u, t, 0.0523674852303482457616113);
294  u = vfmadd(u, t, 0.0587666392926673580854313);
295  u = vfmsub(u, t, 0.0666573579361080525984562);
