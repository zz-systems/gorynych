#pragma once

#include <thread>
#include <mutex>
#include "util/memory.hpp"

namespace zacc {

    template<typename Base, typename Concrete>
    class cloneable : public Base {
    public:
        FORWARD(cloneable);

        virtual ~cloneable() {
        }

        virtual std::shared_ptr<Base> clone() const {
            return zacc::make_shared<Concrete>(static_cast<Concrete const &>(*this));
        }
    };
}
